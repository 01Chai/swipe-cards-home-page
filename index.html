// POST /rent-car
app.post('/translate', (req, res) => {
  const { carModel, descText, lang } = req.body;
  translate(descText, { to: lang }).then(result => {
    res.json({
      background: `url(${carModel}.jpg)`,
      translatedDesc: result.text,
      cta: 'Rent luxury now'
    });
  });
});
